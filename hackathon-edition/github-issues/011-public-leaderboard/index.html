    Public Leaderboard and Project Gallery

    Overview

    Create a public-facing website where people can browse hackathon projects, see scores, and watch episodes.

    Background

    The public needs a way to explore submitted projects, understand the scoring, and watch the AI judges evaluate each project.

    Requirements

    Leaderboard Features

    1. Ranked Project List
    2. Sort by final score
    3. Filter by category
    4. Search by name/team

    5. Show score breakdowns

    6. Project Details Page

    7. Full project information
    8. Score breakdown by judge
    9. Community reaction counts
    10. Embedded YouTube episode

    11. Links to GitHub/demo

    12. Category Views

    13. Projects grouped by category
    14. Category winners highlighted

    15. Statistics per category

    16. Judge Insights

    17. Aggregate scoring patterns
    18. Judge commentary highlights
    19. Scoring criteria explanation

    API Endpoints

    # Public API (read-only)
@app.route('/api/public/leaderboard')
def public_leaderboard():
    category = request.args.get('category')
    limit = request.args.get('limit', 50)

    projects = get_ranked_projects(category, limit)

    # Sanitize data for public consumption
    public_data = []
    for project in projects:
        public_data.append({
            'rank': project['rank'],
            'id': project['id'],
            'name': project['project_name'],
            'team': project['team_name'],
            'category': project['category'],
            'final_score': project['final_score'],
            'youtube_url': project.get('youtube_url'),
            'github_url': project['github_url']
        })

    return jsonify(public_data)

@app.route('/api/public/project/<project_id>')
def public_project_details(project_id):
    project = get_project_details(project_id)

    if not project:
        return jsonify({'error': 'Project not found'}), 404

    # Include detailed scoring
    return jsonify({
        'project': sanitize_project_data(project),
        'scores': get_public_scores(project_id),
        'community': get_community_stats(project_id),
        'episode': get_episode_info(project_id)
    })

    Frontend Components

    // Leaderboard Component
class HackathonLeaderboard {
    constructor(containerId) {
        this.container = document.getElementById(containerId);
        this.currentFilter = 'all';
        this.projects = [];
    }

    async loadProjects() {
        const response = await fetch(`/api/public/leaderboard?category=${this.currentFilter}`);
        this.projects = await response.json();
        this.render();
    }

    render() {
        const html = `
            <div class="leaderboard">
                ${this.renderFilters()}
                <table class="project-table">
                    <thead>
                        <tr>
                            <th>Rank</th>
                            <th>Project</th>
                            <th>Team</th>
                            <th>Category</th>
                            <th>Score</th>
                            <th>Episode</th>
                        </tr>
                    </thead>
                    <tbody>
                        ${this.projects.map(p => this.renderProject(p)).join('')}
                    </tbody>
                </table>
            </div>
        `;
        this.container.innerHTML = html;
    }

    renderProject(project) {
        return `
            <tr class="project-row" onclick="viewProject('${project.id}')">
                <td class="rank rank-${project.rank}">#${project.rank}</td>
                <td class="project-name">${project.name}</td>
                <td>${project.team}</td>
                <td><span class="category-badge ${project.category}">${project.category}</span></td>
                <td class="score">${project.final_score.toFixed(1)}</td>
                <td>
                    ${project.youtube_url ? 
                        `<a href="${project.youtube_url}" target="_blank">Watch</a>` : 
                        'Coming Soon'}
                </td>
            </tr>
        `;
    }
}

    Caching Strategy

    from flask_caching import Cache

cache = Cache(app, config={'CACHE_TYPE': 'simple'})

@cache.cached(timeout=300)  # Cache for 5 minutes
def get_ranked_projects(category=None, limit=50):
    # Expensive database query
    query = """
        SELECT p.*, 
               (SELECT MAX(weighted_total) FROM hackathon_scores 
                WHERE submission_id = p.submission_id AND round = 2) as final_score
        FROM pitches p
        WHERE p.status = 'completed'
        {}
        ORDER BY final_score DESC
        LIMIT ?
    """.format("AND p.category = ?" if category else "")

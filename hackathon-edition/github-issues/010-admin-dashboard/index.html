    Create Admin Dashboard

    Overview

    Build a simple admin interface to manage hackathon submissions, monitor scoring progress, and control the pipeline.

    Background

    Admins need visibility into submission status, scoring progress, and ability to trigger various pipeline stages manually.

    Requirements

    Dashboard Features

    1. Submission Management
    2. View all submissions with status
    3. Trigger research for specific projects
    4. Mark submissions for review

    5. Export submission data

    6. Scoring Overview

    7. View Round 1 scores
    8. Monitor Discord voting
    9. Trigger Round 2 scoring

    10. See final rankings

    11. Pipeline Control

    12. Generate episodes for selected projects
    13. Queue recordings
    14. Monitor upload status

    15. Retry failed operations

    16. Analytics

    17. Submission statistics by category
    18. Score distributions
    19. Community engagement metrics
    20. Pipeline performance

    # app.py - Flask application
from flask import Flask, render_template, jsonify, request
import sqlite3

app = Flask(__name__)

@app.route('/')
def dashboard():
    stats = get_dashboard_stats()
    return render_template('dashboard.html', stats=stats)

@app.route('/api/submissions')
def get_submissions():
    status_filter = request.args.get('status', 'all')
    submissions = fetch_submissions(status_filter)
    return jsonify(submissions)

@app.route('/api/trigger/research/<submission_id>', methods=['POST'])
def trigger_research(submission_id):
    # Queue research job
    result = queue_research_job(submission_id)
    return jsonify({'status': 'queued', 'job_id': result})

@app.route('/api/scores/<submission_id>')
def get_scores(submission_id):
    scores = fetch_all_scores(submission_id)
    return jsonify(scores)

    def get_submission_status(submission_id):
    """Get comprehensive status for a submission"""

    conn = sqlite3.connect('hackathon.db')

    status = {
        'submission': get_submission_data(conn, submission_id),
        'research': check_research_status(conn, submission_id),
        'round1_scoring': check_round1_status(conn, submission_id),
        'community_voting': get_voting_stats(conn, submission_id),
        'round2_scoring': check_round2_status(conn, submission_id),
        'episode_generation': check_episode_status(conn, submission_id),
        'recording': check_recording_status(conn, submission_id),
        'upload': check_upload_status(conn, submission_id)
    }

    conn.close()
    return status

    Pipeline Control

    @app.route('/api/pipeline/control', methods=['POST'])
def pipeline_control():
    action = request.json.get('action')
    submission_ids = request.json.get('submission_ids', [])

    actions = {
        'research': lambda ids: batch_research(ids),
        'score_round1': lambda ids: batch_score_round1(ids),
        'score_round2': lambda ids: batch_score_round2(ids),
        'generate_episodes': lambda ids: batch_generate_episodes(ids),
        'record': lambda ids: batch_record(ids),
        'upload': lambda ids: batch_upload(ids)
    }

    if action in actions:
        job_id = actions[action](submission_ids)
        return jsonify({'status': 'started', 'job_id': job_id})

    return jsonify({'error': 'Invalid action'}), 400
